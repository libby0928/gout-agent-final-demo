  const content = fs.readFileSync(filePath, 'utf-8');
  const bodyContent = content.replace(/<style[^>]*>[\s\S]*?<\/style>/gi, '')
                             .replace(/<script[^>]*>[\s\S]*?<\/script>/gi, '');
  const issues = [];
  for (const word of FORBIDDEN_WORDS) {
    if (bodyContent.includes(word)) issues.push(`禁用词: "${word}"`);
  }
  for (const code of FORBIDDEN_CODES) {
    if (bodyContent.includes(code)) issues.push(`字段编码泄露: "${code}"`);
  }
  return issues;
}
